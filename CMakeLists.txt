cmake_minimum_required(VERSION 3.27)

project(ProjectA)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})

add_subdirectory(GameEngine)
add_subdirectory(Dodge3D)

set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT Dodge3D)

set(CPACK_GENERATOR "NSIS")
set (VC_REDIST_PATH "${CMAKE_CURRENT_SOURCE_DIR}/GameEngine/Bin/VC_redist.x64.exe")

install(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/GameEngine/Shader DESTINATION GameEngine)
install(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/Dodge3D/Resource DESTINATION Dodge3D)
install(PROGRAMS "${VC_REDIST_PATH}" DESTINATION Bin COMPONENT vcredist)
install(TARGETS Dodge3D DESTINATION .)

set(CPACK_PACKAGE_NAME "Dodge3D")
set(CPACK_PACKAGE_VERSION "0.0.0")
set(CPACK_PACKAGE_VENDOR "Github-ChoiJiOne")
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "https://github.com/ChoiJiOne/ProjectA")
set(CPACK_PACKAGE_INSTALL_DIRECTORY "Dodge3D")
set(CPACK_PACKAGE_EXECUTABLES "Dodge3D" "Dodge3D")

set(CPACK_NSIS_EXTRA_INSTALL_COMMANDS "${CPACK_NSIS_EXTRA_INSTALL_COMMANDS} ExecWait '\\\"$INSTDIR\\\\Bin\\\\VC_redist.x64.exe\\\" /install /q'")

set(CPACK_RESOURCE_FILE_LICENSE "${CMAKE_CURRENT_SOURCE_DIR}/LICENSE.txt")

include(CPack)
